#%%
import pandas as pd 
import altair as alt
  
# data import  
url = "https://github.com/byuidatascience/data4python4ds/raw/master/data-raw/mpg/mpg.csv"
  
# Reads csv file
mpg = pd.read_csv(url)
  
# Stores JSON data 
alt.data_transformers.enable('json')
  
#Creates chart 
chart = (alt.Chart(mpg)
  .encode(
    x = "displ",
    y = "hwy",
    color = "class"
    )
  .mark_circle())
chart.save('chart.png')
